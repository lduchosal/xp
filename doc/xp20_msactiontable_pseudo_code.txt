

  func list[str] generateMSTableTelegrams(str data) :

    str data = ""
    remove starting AAAA

    byte shortLong = bitsToByte(inputLogicSetup.shortLong)
    data = data + Checksum.nibble(shortLong)

    groupOnOff = bitsToByte(inputLogicSetup.groupOnOff)
    data = data + Checksum.nibble(groupOnOff)

    invert = bitsToByte(inputLogicSetup.invert)
    data = data + Checksum.nibble(invert)

    for (int n = 0 n < 8 n++) :
      channel = bitsToByte(inputLogicSetup.andFunctions(n))
      data = data + Checksum.nibble(channel)

    tempByte = bitsToByte(this.SA_funcSelect)
    data = data + Checksum.nibble(tempByte)


    tempByte = bitsToByte(this.TA_funcSelect)
    data = data + Checksum.nibble(tempByte)

    int msBlockSize = 32
    int msXpSensorMsLength = 13
    for (int i = msXpSensorMsLength i < msBlockSize i++) :
      data = data + "AA"

    return data
  }
